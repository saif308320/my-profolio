<script>
    const slides = document.querySelectorAll('.slide');
    const dotsContainer = document.querySelector('.dots');
    let current = 0;
    let isAnimating = false;

    // Create Dots
    slides.forEach((_, i) => {
        const dot = document.createElement('div');
        dot.className = `dot ${i === 0 ? 'active' : ''}`;
        dot.addEventListener('click', () => { if(!isAnimating) move(i); });
        dotsContainer.appendChild(dot);
    });

    const dots = document.querySelectorAll('.dot');

    function move(nextIndex) {
        if (isAnimating || nextIndex === current) return;
        isAnimating = true;

        const outgoing = slides[current];
        const incoming = slides[nextIndex];

        // GSAP Power Animation
        const tl = gsap.timeline({
            onComplete: () => {
                outgoing.classList.remove('active');
                isAnimating = false;
                current = nextIndex;
            }
        });

        // Current Slide Out
        tl.to(outgoing.querySelector('.testimonial-text'), { opacity: 0, y: -20, duration: 0.4 })
          .to(outgoing.querySelector('.image-wrapper'), { scale: 0.9, opacity: 0, duration: 0.5 }, "-=0.3")
          .set(outgoing, { display: 'none' });

        // New Slide In
        tl.set(incoming, { display: 'flex', opacity: 1 })
          .fromTo(incoming.querySelector('.image-wrapper'), 
            { x: 100, opacity: 0, rotateY: -20 }, 
            { x: 0, opacity: 1, rotateY: 0, duration: 0.8, ease: "power2.out" })
          .fromTo(incoming.querySelector('.testimonial-text'), 
            { y: 30, opacity: 0 }, 
            { y: 0, opacity: 1, duration: 0.6, ease: "power2.out" }, "-=0.4")
          .fromTo(incoming.querySelector('.author-info'), 
            { opacity: 0 }, { opacity: 1, duration: 0.4 });

        dots.forEach((d, i) => d.classList.toggle('active', i === nextIndex));
        incoming.classList.add('active');
    }

    document.getElementById('next').onclick = () => {
        move((current + 1) % slides.length);
    };

    document.getElementById('prev').onclick = () => {
        move((current - 1 + slides.length) % slides.length);
    };
</script>

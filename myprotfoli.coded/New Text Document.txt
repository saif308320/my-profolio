<script>
  lucide.createIcons();
  AOS.init({ duration: 1000, once: true });

  /* --- AURA CURSOR LOGIC --- */
  const dot = document.querySelector('.cursor-dot');
  const outline = document.querySelector('.cursor-outline');
  let mouseX = 0, mouseY = 0, dotX = 0, dotY = 0, outX = 0, outY = 0;

  document.addEventListener('mousemove', e => {
    mouseX = e.clientX;
    mouseY = e.clientY;
  });

  function animateCursor() {
    dotX += (mouseX - dotX) * 0.2;
    dotY += (mouseY - dotY) * 0.2;
    dot.style.left = dotX + 'px';
    dot.style.top = dotY + 'px';
    outX += (mouseX - outX) * 0.12;
    outY += (mouseY - outY) * 0.12;
    outline.style.left = outX + 'px';
    outline.style.top = outY + 'px';
    requestAnimationFrame(animateCursor);
  }
  animateCursor();

  /* --- GLOW EFFECT FOR CARDS --- */
  const glowCards = document.querySelectorAll('.service-card, .skill-card, .stat-item');
  glowCards.forEach(card => {
    card.addEventListener('mousemove', (e) => {
      const rect = card.getBoundingClientRect();
      const x = ((e.clientX - rect.left) / rect.width) * 100;
      const y = ((e.clientY - rect.top) / rect.height) * 100;
      card.style.setProperty('--x', `${x}%`);
      card.style.setProperty('--y', `${y}%`);
    });
  });

  const auraTargets = document.querySelectorAll('a, button, .service-card, .skill-card, .stat-item, .logo');
  auraTargets.forEach(t => {
    t.addEventListener('mouseenter', () => document.body.classList.add('cursor-hover'));
    t.addEventListener('mouseleave', () => document.body.classList.remove('cursor-hover'));
  });

  /* --- Particles Config --- */
  particlesJS('particles-js', {
    "particles": {
      "number": { "value": 80, "density": { "enable": true, "value_area": 800 } },
      "color": { "value": "#0088FF" },
      "shape": { "type": "circle" },
      "opacity": { "value": 0.2, "random": false },
      "size": { "value": 3, "random": true },
      "line_linked": { "enable": true, "distance": 150, "color": "#0088FF", "opacity": 0.1, "width": 1 },
      "move": { "enable": true, "speed": 2, "direction": "none", "random": false, "straight": false, "out_mode": "out", "bounce": false }
    },
    "interactivity": {
      "events": { "onhover": { "enable": true, "mode": "grab" }, "onclick": { "enable": true, "mode": "push" } },
      "modes": { "grab": { "distance": 140, "line_linked": { "opacity": 1 } } }
    },
    "retina_detect": true
  });

  const textEl = document.getElementById('type-target');
  const items = ["GHL Expert", "Frontend Dev", "Automation Pro"];
  let i = 0, j = 0, isBack = false;

  function typeAnim() {
    const full = items[i];
    textEl.textContent = full.substring(0, j);
    if (!isBack && j < full.length) { j++; setTimeout(typeAnim, 100); }
    else if (isBack && j > 0) { j--; setTimeout(typeAnim, 50); }
    else { isBack = !isBack; if (!isBack) i = (i + 1) % items.length; setTimeout(typeAnim, isBack ? 2000 : 500); }
  }
  typeAnim();
</script>

<script>
    const canvas = document.getElementById('btnCanvas');
    const ctx = canvas.getContext('2d');
    const target = document.getElementById('target');
    let width, height, particles = [];
    let isHovered = false;
    let mouse = { x: 0, y: 0 };

    function init() {
        width = 280; height = 90;
        canvas.width = width; canvas.height = height;
        for(let i=0; i<80; i++) {
            particles.push({
                x: Math.random() * width,
                y: Math.random() * height,
                vx: (Math.random() - 0.5) * 2,
                vy: (Math.random() - 0.5) * 2,
                size: Math.random() * 2 + 1
            });
        }
    }

    function draw() {
        ctx.clearRect(0, 0, width, height);
        
        // Border Logic (Computational)
        ctx.strokeStyle = isHovered ? '#00f2ff' : '#444';
        ctx.lineWidth = 2;
        ctx.strokeRect(5, 5, width-10, height-10);

        particles.forEach(p => {
            p.x += p.vx; p.y += p.vy;

            // Bounce Math
            if(p.x < 5 || p.x > width-5) p.vx *= -1;
            if(p.y < 5 || p.y > height-5) p.vy *= -1;

            // Mouse Interaction (The "Ultra" Part)
            if(isHovered) {
                let dx = mouse.x - p.x;
                let dy = mouse.y - p.y;
                let dist = Math.sqrt(dx*dx + dy*dy);
                if(dist < 60) {
                    p.x -= dx * 0.05; // Particle Attraction
                    p.y -= dy * 0.05;
                    ctx.fillStyle = '#00f2ff';
                } else {
                    ctx.fillStyle = 'white';
                }
            } else {
                ctx.fillStyle = 'rgba(255,255,255,0.3)';
            }

            ctx.beginPath();
            ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
            ctx.fill();

            // Web-link logic (Drawing lines between nearby particles)
            particles.forEach(p2 => {
                let d = Math.sqrt((p.x-p2.x)**2 + (p.y-p2.y)**2);
                if(d < 40) {
                    ctx.strokeStyle = `rgba(0, 242, 255, ${isHovered ? 0.2 : 0})`;
                    ctx.lineWidth = 0.5;
                    ctx.beginPath();
                    ctx.moveTo(p.x, p.y);
                    ctx.lineTo(p2.x, p2.y);
                    ctx.stroke();
                }
            });
        });
        requestAnimationFrame(draw);
    }

    target.onmousemove = (e) => {
        let rect = target.getBoundingClientRect();
        mouse.x = e.clientX - rect.left;
        mouse.y = e.clientY - rect.top;
        isHovered = true;
    };
    target.onmouseleave = () => isHovered = false;

    init(); draw();
</script>
